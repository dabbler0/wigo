    render

    Same as grid render, but with ‘X’ for prize.

      render: ->
    str = ''
    for j in [0...@h]
      for i in [0...@w]
        if @state.layers[0][j * @w + i] is 1
          str += '@'
        else if @state.layers[1][j * @w + i] is 1
          str += '#'
        else if @state.layers[2][j * @w + i] is 1
          str += 'X'
        else
          str += ' '
      str += '\n'
    return str
